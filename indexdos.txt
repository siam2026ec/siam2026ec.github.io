<script>
  document.addEventListener("DOMContentLoaded", () => {
    // Inicializar AOS
    AOS.init({
      duration: 800,
      once: false, // permite animación repetida
    });

    // --- MENÚ MÓVIL ---
    const menuBtn = document.querySelector("#menuBtn");
    const mobileMenu = document.querySelector("#mobileMenu");

    menuBtn.addEventListener("click", () => {
      mobileMenu.classList.toggle("hidden");
    });

    // Cerrar menú al elegir un link
    document.querySelectorAll(".mobile-link").forEach(link => {
      link.addEventListener("click", () => {
        mobileMenu.classList.add("hidden");
      });
    });

    // --- REINICIAR AOS AL NAVEGAR ENTRE SECCIONES ---
    document.querySelectorAll(".nav-link, .mobile-link").forEach(el => {
      el.addEventListener("click", () => {
        setTimeout(() => {
          AOS.refreshHard();   // este SÍ reinicia animaciones duras
        }, 300);
      });
    });
    // Scroll compensado (para que AOS pueda reanimar)
    document.querySelectorAll('a[data-offset]').forEach(link => {
      link.addEventListener("click", (e) => {
        e.preventDefault();
        const targetID = link.getAttribute("href");
        const offset = parseInt(link.getAttribute("data-offset"), 10);

        const element = document.querySelector(targetID);
        const topPos = element.offsetTop - offset;

        window.scrollTo({ top: topPos, behavior: "smooth" });

        setTimeout(() => AOS.refreshHard(), 600);
      });
    });

  });

</script>
